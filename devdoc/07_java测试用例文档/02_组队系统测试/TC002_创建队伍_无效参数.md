// 创建无效参数队伍请求
REQ_CONTROL_GROUP reqControlGroup = new REQ_CONTROL_GROUP();
reqControlGroup.type = 13;
reqControlGroup.area = -1;
reqControlGroup.subtype = -1;
reqControlGroup.stageindex = -1;
reqControlGroup.dungeonindex = -1;
reqControlGroup.minlevel = -1;
reqControlGroup.maxlevel = -1;
reqControlGroup.partyname = "";
reqControlGroup.isplayer = false;

// 发送请求
byte[] reqData = MessageCodec.encodeMessage(reqControlGroup, (byte) CmdID.REQ_CONTROL_GROUP.getValue());
out.write(reqData);
out.flush();

// 读取响应
byte[] respData = readMessage(in);
RES_CONTROL_GROUP resControlGroup = (RES_CONTROL_GROUP) MessageCodec.decodeMessage(respData);
