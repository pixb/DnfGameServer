### 3.1 核心消息结构

#### 3.1.1 好友请求列表

```protobuf
// 好友请求列表请求
message FriendRequestListRequest {
  int32 character_id = 1;       // 角色ID
  int32 page = 2;                // 页码
  int32 page_size = 3;           // 每页数量
}

// 好友请求信息
message FriendRequestInfo {
  int32 request_id = 1;         // 请求ID
  int32 from_character_id = 2;    // 发送者角色ID
  string from_character_name = 3;  // 发送者角色名称
  int32 level = 4;               // 发送者角色等级
  int32 job = 5;                 // 发送者角色职业
  string verify_message = 6;      // 验证消息
  int64 request_time = 7;         // 请求时间
}

// 好友请求列表响应
message FriendRequestListResponse {
  int32 error = 1;              // 错误码
  int32 character_id = 2;       // 角色ID
  repeated FriendRequestInfo requests = 3;  // 好友请求列表
  int32 total_count = 4;         // 总请求数
  int32 page = 5;                // 当前页码
  int32 page_size = 6;           // 每页数量
}
```

```protobuf
// 好友黑名单请求
message FriendBlacklistRequest {
  int32 character_id = 1;       // 角色ID

  int32 target_character_id = 3;  // 目标角色ID
  int32 page = 4;                // 页码（查询时使用）

message FriendBlacklistResponse {

}
```
