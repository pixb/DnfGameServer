### 3.1 核心消息结构

#### 3.1.1 公会创建

```protobuf
// 公会创建请求
message GuildCreateRequest {
  int32 character_id = 1;       // 角色ID
  string guild_name = 2;         // 公会名称
  string guild_notice = 3;       // 公会公告
  int32 guild_type = 4;          // 公会类型
}

// 公会创建响应
message GuildCreateResponse {
  int32 error = 1;              // 错误码
  int32 character_id = 2;       // 角色ID
  int64 guild_id = 3;            // 公会ID
  string guild_name = 4;         // 公会名称
  bool success = 5;              // 是否成功
}
```

#### 3.1.2 公会加入


#### 3.1.3 公会信息

```protobuf
// 公会信息请求
message GuildInfoRequest {
  int32 character_id = 1;       // 角色ID
  int64 guild_id = 2;            // 公会ID
}

// 公会信息响应
message GuildInfoResponse {
  int32 error = 1;              // 错误码
  int32 character_id = 2;       // 角色ID
  GuildInfo guild_info = 3;      // 公会信息
}

// 公会信息
message GuildInfo {
  int64 guild_id = 1;            // 公会ID
  string guild_name = 2;         // 公会名称
  string guild_notice = 3;       // 公会公告
  int32 guild_level = 4;         // 公会等级
  int32 guild_exp = 5;           // 公会经验
  int32 member_count = 6;        // 成员数量
  int32 max_member_count = 7;    // 最大成员数量
  string guild_master = 8;       // 公会会长
  int32 guild_type = 9;          // 公会类型
  int64 create_time = 10;        // 创建时间
}
```

#### 3.1.4 公会列表

```protobuf
// 公会列表请求
message GuildListRequest {
  int32 character_id = 1;       // 角色ID

  string search_name = 4;        // 搜索名称
  int32 sort_type = 5;           // 排序类型

// 公会列表响应
message GuildListResponse {
  int32 error = 1;              // 错误码

}
```
